int main() {
    int sum = 0;
    for (int i = 1; i <= 10; i = i + 1) {
        if (i % 2 == 0) {
            sum = sum + i;
        } else {
            sum = sum - i;
        }
    }
    while (sum < 100) {
        sum = sum + 5;
        cout << "Current sum: " << sum << endl;  // print status
    }
    return sum;
}
